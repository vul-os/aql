# BOT KORP MODERN TECH COLOR PALETTE
# Sleek, modern design with high contrast and striking accents

# === PRIMARY COLOR SCHEME ===

Matte Black: #121212             # Primary - Professional, modern, high-tech
Bright Tangerine: #FF6B35        # Accent - Energy, innovation, action
Slate Blue-Gray: #4F5D75         # Supporting - Stability, trust
Cool Silver: #B0B3B8             # Supporting - Sophistication, neutrality

# === EXTENDED PALETTE ===

# Black Variations
Matte Black: #121212            # Primary black (main brand color)
Black Light: #1a1a1a            # Slightly lighter black for cards
Black Soft: #2a2a2a             # Even lighter for subtle contrast

# Orange Variations  
Bright Tangerine: #FF6B35       # Primary accent (main CTA color)
Tangerine Dark: #E85A2A         # Darker orange for hover states
Tangerine Light: #FF8A5C        # Lighter orange for highlights

# Supporting Color Variations
Slate Blue-Gray: #4F5D75        # Secondary color
Slate Blue Light: #6B7A94       # Lighter slate for subtle elements
Cool Silver: #B0B3B8            # Neutral supporting color
Silver Light: #D0D2D5           # Lighter silver for borders

# === UTILITY COLORS ===

White: #FFFFFF                  # Pure white for text on dark
Off-White: #FAFAFA              # Soft white for light backgrounds
Light Gray: #E5E7EB             # Light borders and dividers

Success: #10B981                # Green for success states
Warning: #F59E0B                # Amber for warnings
Error: #EF4444                  # Red for errors

# === USAGE GUIDELINES ===

# Primary Actions: Use Bright Tangerine (#FF6B35)
# - CTA buttons
# - Important actions
# - Active states
# - Focus indicators

# Primary Backgrounds: Use Matte Black (#121212)
# - Headers/navigation
# - Dark sections
# - Primary cards
# - Hero sections

# Secondary Elements: Use Slate Blue-Gray (#4F5D75)
# - Secondary buttons
# - Interactive elements
# - Supporting text
# - Informational badges

# Neutral Elements: Use Cool Silver (#B0B3B8)
# - Borders
# - Dividers
# - Muted text
# - Disabled states

# Gradients:
# - Hero sections: from-botkorp-black via-botkorp-black-light to-botkorp-slate-blue
# - Glow effects: botkorp-orange with opacity
# - Brand gradient: from-botkorp-orange to-botkorp-slate-blue

# === DARK MODE ===
# Dark mode uses inverted hierarchy:
# - Background: Matte Black (#121212)
# - Primary: Bright Tangerine (#FF6B35)
# - Text: White (#FFFFFF)
# - Accents: Cool Silver (#B0B3B8)
